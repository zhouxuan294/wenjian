隔行变色（两行一组）：

1、用css实现：

<style type='text/css'>

    ul{

        margin:auto;

        width:500px;

        list-style:none;

        }

    li{

        width:100%;

        height:50px;

        text-align:center;

        font-size:25px;

        border:1px solid 颜色;

        }

    li:nth-child(2n){

        background-color:颜色;

        }

    li:nth-child(2n-1){

        background-color:颜色;

        }

</style>


html部分：

<ul>

    <li>1</li>

    <li>2</li>

    <li>3</li>

    <li>4</li>

    <li>5</li>

    <li>6</li>

</ul>


2、用js实现：
html结构与css方法的相同

<style type='text/css'>

    ul{

        margin:auto;

        width:500px;

        list-style:none;

        }

    li{

        width:100%;

        height:50px;

        text-align:center;

        font-size:25px;

        border:1px solid 颜色;

        }

以下方法共用此css

  • 方法一(不推荐)：
var oLis=document.getElementsByTagName('li');

oLis[0].style.backgroundColor='颜色1';

oLis[1].style.backgroundColor='颜色2';

oLis[2].style.backgroundColor='颜色1';

oLis[3].style.backgroundColor='颜色2';

oLis[4].style.backgroundColor='颜色1';

oLis[5].style.backgroundColor='颜色2';

  • 方法二（for循环+if条件）：
var oLis=document.getElementsByTagName('li');

for(var i=0;i<oLis.length;i++){

    if(i%2==0){

        oLis[i].style.backgroundColor='颜色1';

        }else{

        oLis[i].style.backgroundColor='颜色2';

        }

    }

  • 方法三（for循环，仅适用于行数为偶数时）：
var oLis=document.getElementsByTagName('li');

for(var i=1;i<oLis.length;i+=2){

    oLis[i-1].style.backgroundColor='颜色1';

    oLis[i].style.backgroundColor='颜色2';

    }

  • 方法四（function函数+for循环+if条件）：
var oLis=documetnt.getElementsByTagName('li');

function changeBg(a){

    for(var i=0;i<a.length;i++){

        if(i%2==0){

            a[i].style.backgroundColor='颜色1';

            }else{

            a[i].style.backgroundColor='颜色2';

            }

        }

    }

changeBg(oLis);

  • 方法五（for循环+switch条件）：
var oLis=document.getElementsByTagName('li');

for(var i=0;i<oLis.length;i++){

    switch(i%2){

        case 0:

            oLis[i].style.backgroundColor='颜色1';

            break;

        case 1(此处也可以写default，同时省略break):

            oLis[i].style.backgroundColor='颜色2';

            break;

            }

        }

  • 方法六（function函数+for循环+switch条件）：
var oLis=document.getElementsByTagName('li');

function changeBg(){

     for(var i=0;i<oLis.length;i++){

         switch(i%2){

             case 0:

                 oLis[i].style.backgroundColor='颜色1';

                 break;

             case 1(此处也可以写default，同时省略break):

                 oLis[i].style.backgroundColor='颜色2';

                 break;

             }

         }

     }

changeBg();
