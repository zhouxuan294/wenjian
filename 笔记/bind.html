bind

使函数的this指向第一个参数，返回的是this指向改变后的函数，新函数执行时，再传参，会把参数与bind传的参数（除第一个参数外）合并。

1.function sum(){

2.   console.log(this);

3.   console.log(arguments);

4.}

5.function sum2(){

6.    return 2;

7.}

8.var sum3=sum.bind(sum2,1,2,3);

9.sum3();//f sum2(){...} [1,2,3]

10.sum(1,2,3);//f sum2{...} [1,2,3,1,2,3]

1.function myBind(){

2.   var _this=arguments[0];

3.   var that=this;

4.   var ary=[];

5.   for(var i=1;i<arguments.length;i++){

6.      ary.push(arguments[i]);

7.   }

8.   return function(){

9.      var arr=[];

10.      for(var i=0;i<arguments.length;i++){

11.         arr.push(arguments[i]);

12.      }

13.      that.apply(_this,ary.concat(arr));

14.   }

15.}

17.function myBind(...arr){

18.   var that=arr.shift();

21.      _this.apply(that,arr.concat(ary));

22.   }

23.}
