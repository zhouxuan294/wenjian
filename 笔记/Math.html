2.Math.pow(3,2);//9;3的2次方

3.Math.floor(2.9);//2;小数向下取整，作用等同于parseInt（正数）

4.Math.ceil(2.1);//3;小数向上取整，作用等同于pardeInt(负数)

5.Math.abs(-2);//2;取绝对值

6.Math.round(2.1);//2;四舍五入

7.Math.random();//取[0,1)之间的小数，可以取到0，但取不到1

8.Math.max(1,2,3);//3;取一组数中的最大值

9.Math.min(1,2,3);//1;取一组数中的最小值

10.Math.round(Math.random()*(m-n)+n);//取[n,m]之间的随机整数
