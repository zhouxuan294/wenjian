数组的排列和排序（会改变原数组）：

  • reverse：数组的反转；没有参数；返回值为反转后的数组。
var ary=[1,2,3,4,5,6];

var a=ary.reverse();

console.log(a);//[6,5,4,3,2,1]

console.log(ary);//[6,5,4,3,2,1]

a[0]=9;

console.log(a);//[9,2,3,4,5,6]

console.log(ary);//[9,2,3,4,5,6]


用function函数实现reverse功能：
1）交换第一项和最后一项、第二项和倒数第二项…的位置：

var ary=[1,2,3,4,5,6];

function myReverse(ary){

    for(var i=0;i<ary.length;i++){

        var temp=ary[i];

        ary[i]=ary[ary.length-1-i];

        ary[ary.length-1-i]=temp;

        }

    return ary;

    }

myReverse(ary);//[6,5,4,3,2,1]


2）建立一个空数组，再利用push（在数组的最后添加一项）从原数组的最后一项开始往前，将原数组的每一项添加到新数组中。

var ary=[1,2,3,4,5,6];

function myReverse(ary){

    var arr=[];

    for(var i=ary.length-1;i>=0;i--){

        arr.push(ary[i]);

        }

    return arr;

    }

myReverse(ary);//[6,5,4,3,2,1]

  • sort：将数组排序（默认从小到大）；参数可有可无，没有参数时，只能实现10以内的排序；有参数时，参数是一个函数（回调函数）；返回值是排序完成的数组，与原数组是一个地址。
var ary=[1,2,3,1,2,3,4,5,6];

var a=ary.sort();

console.log(a);//[1,1,2,2,3,3,4,5,6]

console.log(ary);//[1,1,2,2,3,3,4,5,6]

a[0]=9;

console.log(a);//[91,2,2,3,3,4,5,6]

console.log(ary);//[91,2,2,3,3,4,5,6]

var ary2=[1,2,3,4,15,51,6];

var ary3=[1,2,3,1,2,3,4,5,6,55,66,22];

（顺序为从小到大）

var b=ary3.sort(function(a,b){

    return a-b;//返回值为正数时，对应两项交换位置

    });

console.log(b);//[1,1,2,2,3,3,4,5,6,22,55,66]

（顺序为从大到小）

var c=ary3.sort(function(a,b){

    return b-a;//返回值为正数时，对应两项交换位置

    })

console.log(c);//[66,55,22,6,5,4,3,3,2,2,1,1]


用function实现sort的功能：
（从小到大）

function sum(ary){

    for(var i=0;i<ary.length;i++){

        for(n=i;n<ary/length;n++){

            var b=ary[i]-ary[n];

            if(b>0){

                var temp=ary[i];

                ary[i]=ary[n];

                ary[n]=temp;

                }

            }

        }

    return ary;

    }

sum(ary);


（从大到小）
若要实现将数组从大到小排列，只需将上述代码中的var b=ary[i]-ary[n]换成var b=ary[n]-ary[i]即可，或者将if(b>0)改为if(b<0)即可。
