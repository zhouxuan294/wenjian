Vue-组件间传递数据-父级传给子级

1.<div id='app'>

2.    <demo_2></demo_2>

3.</div>

1.<!--子级组件-->

2.<template id='test_1'>

3.    <div>

4.        <button @click='fn_1'>子级组件的按钮</button>

5.        {{n}}

6.        {{s}}

7.        {{a}}

8.        {{o}}

9.    </div>

10.</template>

11.<!--父级组件-->

12.<template id='test_2'>

13.    <div>

14.        <button @click='fn_2'>父级组件的按钮</button>

15.        {{num}}

16.        {{str}}

17.        {{ary}}

18.        {{obj}}

19.        <demo_1 :s='str' :a='ary' :n='num' :o='obj'></demo_1><!--此处若没有“:”，则它们仅仅是字符串，子级组件获取到的只是'str','ary','num','obj'；有“:”时，它们才是变量，子级组件获取到的才是它们各自对应的值-->

20.    </div>

21.</template>

1.//———子级组件———

2.let demo_1={

3.    tempalte:'#test_1',

4.    data(){

5.        return {};

6.    },

7.    props:{

8.        n:{

9.            type:[Array,String,Number,Object],//设置接收的数据类型，只接收一种数据类型的时候：type:Number；接收所有数据类型的时候：type:null，或者不设置

10.            default:'100',//设置默认值，当父级组件没有传递该数据的时候，显示此处设置的值

11.            required:true,//设置是否要求父级组件传递该数据，true是要求，false是不要求；若不写，则默认是不要求

12.            validator(val){//判断传递的数据书否符合要求，符合要求就输出，不符合要求就报错【[Vue warn]: Invalid prop: custom validator check failed for prop "n".】；若不写，则默认都符合要求

13.                console.log(val);//val是父级组件传递的属性的属性值

14.                return true;//这里是判断条件；若不写，则默认都不符合要求

15.            }

16.        },

17.        a:{

18.            type:Array,

19.            default:function(){

20.                return [7,8,9];//如果设置的默认值是引用数据类型，则要用一个函数return这个默认值

21.            },

22.            validator(val){

23.                console.log(val);

24.                return true;

25.            }

26.        },

27.        o:{

28.            default:function(){

29.                return {name:'DEMO'}

30.            },

31.            validator(val){

32.                console.log(val);

33.                return true;

34.            }

35.        },

36.        s:{

37.            default:'STRING',

38.            validator(val){

39.                console.log(val);

40.                return true;

41.            }

42.        }

43.    },

44.    methods:{

45.        fn_1(){

46.            console.log(Object.prototype.toString.call(this.n));//[object Number]

47.            console.log(Object.prototype.toString.call(this.s));//[object String]

48.            console.log(Object.prototype.toString.call(this.a));//[object Array]

49.            console.log(Object.prototype.toString.call(this.o));//[object Object]

50.        }

51.    }

52.};

53.//———父级组件———

54.let demo_2={

55.    template:'#test_2',

56.    data(){

57.        return {

58.            num:0,

59.            str:'string',

60.            ary:[1,2,3,4,5,6],

61.            obj:{name:'TEST'}

62.        }

63.    },

64.    components:{

65.        demo_1,

66.    },

67.    methods:{

68.        fn_2(){

69.            this.num++;

70.        }

71.    }

72.},

73.let vm=new Vue({

74.    el:'#app',

75.    components:{

76.        demo_2,

77.    }

78.})

注意：不要通过子级组件直接更改父级组件传递过来的数据，若要更改数据，应该将对应的数据复制一份，在设置一个变量来接收。例如：父级组件传递的数据是一个 ary，则应该设置一个变量 arr，使 arr=ary.slice()
