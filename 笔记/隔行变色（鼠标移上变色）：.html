一、用css实现：

css部分：

<style type='text/css'>

    *{

        margin:0;

        padding:0;

        list-style:none;

        }

    ul{

        width:400px;

        margin:auto;

        }

    ul li{

        width:100%;

        height:50px;

        line-height:50px;

        text-align:center;

        border-bottom:1px solid 颜色;

        }

    ul li:nth-child(2n){

        background-color:颜色1;

        }

    ul li:nth-child(2n-1){

        background-color:颜色2;

        }

    ul li:hover{

        background-color:颜色3;

        }

</style>

function byTag(tag){

    var oLis=document.getElementsByTagName(Tag);

    return oLis;

    }

var lis=byTag('li');

function changeBg(oLis){

    for(var i=0;i<oLis.langth;i++){

        if(i%2==0){

            oLis[i].style.backgroundColor='颜色1';

            }else{

            oLis[i].style.backgroundColor='颜色2';

            }

        }

    }

changeBg(lis);
