Node-fs（文件系统）

fs.readFile(path,[options],callback)
异步读取文件

  • path【文件路径】
  • options【编码格式，默认是buffer，可以设置为“utf8”】
  • callback【数据读取完成后执行的回调函数】
1.let fs=require("fs");

2.fs.readFile('./custom.json','utf8',function(err,data){

3.    console.log(err);//读取数据出错时的错误信息，没有出错时就是“null”

4.    console.log(data);//读取到的数据:[{"name":"文字"}]

5.})

fs.readFileSync(path,[options])
同步读取数据

  • path【文件路径】
  • options【编码格式】
  • 有一个返回值，就是读取到的数据
1.let fs=require('fs');

2.let test=fs.readFileSync('./custom.json','utf8');

3.console.log(test);//[{"name":"文字"}]

fs.writeFile(path,data,options,callback)
异步写入数据

  • path【文件路径，若该文件存在，就将原文件内容替换；若不存在，就新建】
  • data【要写入的数据，为 String 或者 buffer 类型的数据】
  • options【编码格式，可以不写，默认为buffer类型，可以设置问“ utf8 ”】
  • callback【写入数据成功后执行的函数】
1.let fs=require('fs');

2.fs.writeFile('./custom.json','123','utf8',function(){

3.    console.log("OK");

4.});

5.//若果不想原文件内容被替换，只是添加数据，则应该先读出数据再写入：

6.let fs=require('fs');

7.fs.readFile('./custom.json','utf8',function(err,data){

8.    let ary=JSON.parse(data);

9.    ary.push({"name":"内容"});

10.    data=JSON.stringify(ary);

11.    fs.writeFile('./custom.json',data,'utf8',function(){

12.        console.log("OK");

13.    });

14.})
