数组的增加-删除-修改：

一、增加：
1）

var ary=[];

ary[0]='1';

ary[1]='2';

ary[2]='3';


2）push()：在数组的末尾增加项。push(xxx,xxx,xxx)，参数可以有很多个，且数据类型不限。返回值是改变后的数组的长度。

var ary=['a','b','c'];

var a=ary.push('d');

console.log(ary);//ary=['a','b','c','d']

console.log(a);//4

  • 当push里有函数时：
ary.push(function(){console.log(1)},'d','e');

调用其中函数的方法：

ary[0]();

  • 也可以用数组的属性length来实现push的作用：
var ary=['a','b','c'];

ary[ary.length]='d';

console.log(ary);//ary=['a','b','c','d']

  • 用length写一个函数实现push的作用（仅添加一项）：
var ary=['a','b','c']

function myPush(ary,n){

    ary[ary.length]=n;

    return ary.length;

    }

var a=myPush(ary,'d');

console.log(a);//4

console.log(ary);//ary=['a','b','c','d']

  • 用length写一个函数实现push的作用（添加多项）：
var ary=[1,2,3];

function sum(){

    var a=arguments[0];

    for(var i=1;i<arguments.length;i++){

        a[a.length]=arguments[i];

        }

    return a.length;

    }

sum(ary,4,5,6);//6

console.log(ary);//[1,2,3,4,5,6]

3）unshift：在数组的前边添加项。参数可以是多个；返回值是改变后的数组的长度。

var ary=['a','b','c'];

var a=ary.unshift('d');

console.log(a);//4

console.log(ary);//ary=['d','a','b','c']


二、删除：
1）pop：删除数组的最后一项。没有参数；返回值是被删除的那一项。

var ary=[1,2,3];

var a=ary.pop();

console.log(a);//3(数据类型为number)

console.log(ary);//ary=[1,2]

  • 用length删除数组的最后一项：
var ary=[1,2,3];

ary.length=ary.length-1;(或者：ary.length -= 1)

console.log(ary.length);//2

console.log(ary);//ary=[1,2]

  • 用length写一个函数实现pop的作用：
var ary=[1,2,3];

functiom myPop(ary){

    ary.length-=1;

    return ary.length;

    }

var a=myPop(ary);

console.log(a);//2

console.log(ary);//ary=[1,2]


2）、shift：删除数组的第一项。没有参数；返回值为被删除的那一项。

var ary=[1,2,3];

var a=ary.shift();

console.log(a);//1(数据类型为number)

console.log(ary);//ary=[2,3]

  • 用函数实现shift的功能：
function myShift(ary){

    var a=ary[0];

    for(var i=0;i<ary.length;i++){

        ary[i]=ary[i+1];

        }

    ary.length-=1;

    return a;

    }

3）、splice（n,m,x,y,z...）：从索引为n（包含）的地方开始删除m个元素，用x，y，z代替；返回值是被删除的m项元素组成的新数组。

var ary=[1,2,3,4,5,6];

var a=ary.splice(1,2,'a','b','c');

console.log(a);//[2,3]

console.log(ary);//ary=[1,'a','b','c',4,5,6]

  • 两个特殊情况：
var ary=[1,2,3,4,5,6];

var a=ary.splice();

console.log(a);//[](空数组)

console.log(ary);//原数组不变


var ary=[1,2,3,4,5,6];

var a=ary.splice(0);

console.log(a);//[1,2,3,4,5,6]

console.log(ary);//[](原数组变为空数组)


拓展：

  • 用splice实现push：
var ary=[1,2,3,4,5];

function a(ary,n){

    ary.splice(ary.length,0,n);

    return ary.length;

    }

    console.log(a(ary,7));//6

    console.log(ary);//[1,2,3,4,5,7]

  • 用splice实现pop：
var ary=[1,2,3,4,5];

function a(ary){

    var b=ary[ary.length];

    ary.splice(ary.length-1,1);

    return b;

    }

console.log(a(ary));//5

console.log(ary);//[1,2,3,4]


或者：

var ary=[1,2,3,4,5];

function a(ary){

    var temp=null;

    temp=ary.splice(ary.length-1,1);

    return temp[0];

    }

console.log(a(ary));//5

console.log(ary);//[1,2,3,4]
