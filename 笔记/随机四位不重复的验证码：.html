随机四位不重复的验证码：

<div id=div1>文字内容</div>

<script>

function sum1(){

    var ary=[];

    for(var i=0;i<4;i++){

        var n=Math.round(Math.random()*61);//从0-61中获取4个不同的整数

        if(ary.indexOf(n)==-1){

            ary.push(n);

            }else{

            i--;

            }

        }

    return ary;

    }

function sum2(){

    var str='';

    var str2='qwertyuiopasdfghjklzxcvbnmQWERTYUIOPASDFGHJKLZXCVBNM0123456789';

    var a=sum1();

    for(var i=0;i<a.length;i++){

        str+=str2[a[i]];//将之前获取的4个不同的整数作为str2的索引值，从str2中获取4个不同的字符

        }

    return str;

    }

//将验证码显示在页面中

var a=document.getElementById('div1');

a.onclick=function(){

    a.innerHTML=sum2();

    }
