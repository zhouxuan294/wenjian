数组平均值：

  • 方法一：
1.var ary=[1,2,3,45,56,7,8,9];

2.//去除一个最大值，去除一个最小值，求平均值

3.ary.sort(function(a,b){

4.   return a-b;

5.});

6.ary.shift();

7.ary.pop();

8.var num=eval(ary.join('+'))/ary.length;

9.num.toFixed(2);//设置保留小数点后2位小数，返回的结果是字符串
  • 方法二：rest运算符...，把逗号隔开的值序列组合成一个数组
1.function fn(...args){

2.   console.log(args);//[1,2,3]

3.   args.sort((a,b)=>{return a-b;});

4.   args.shift();

5.   args.pop();

6.   return eval(args.join('+'))/args.length;

7.}

8.fn(1,2,3);
  • 方法三：做一个空数组，把arguments中的数据放到数组里
1.function fn(){

2.   var ary=[];

3.   for(let i=0;i<arguments.length;i++){

4.      ary.push(arguments[i]);

5.   }

6.   ary.sort((a,b)=>{return a-b});

7.   ary.shift();

8.   ary.pop();

9.   return eval(ary.join('+'))/ary.length;

10.}

11.fn(1,2,3);
  • 方法四：将arguments的__proto__指向数组的原型对象
1.function fn(){

2.   arguments.__proto__=Array.prototype;

3.   arguments.sort((a,b)=>{return a-b});

4.   arguments.shift();

5.   arguments.pop();

6.   return eval(arguments.join('+'))/arguments.length;

7.}

8.fn(1,2,3);
  • 通过call或者apply让类数组使用数组的方法
1.function fn(){

2.   var ary=[].slice.call(arguments);//利用slice把arguments的复制一份并转为数组

3.   ary.sort((a,b)=>{return a-b});

4.   ary.shift();

5.   ary,pop();

6.   return eval(ary.join('+'))/ary.length;

7.}

8.fn(7,8,9);
