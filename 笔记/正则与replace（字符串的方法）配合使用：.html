4.str.replace(reg,',');//"qwe,asd456zxc789" 只能替换第一处匹配到的内容,返回的是字符串

5.str.replace(reg,function(){console.log(arguments);});//"qweundefinedasd456zxc789"

2.var reg=/\d+/g;

3.str.replace(reg,function(){

4.   var a=arguments[0]-100;
