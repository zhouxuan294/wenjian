Vue-router-编程式路由切换

1.<div id='app'>

2.    <router-link :to='{path:'/home',query:{a:1,b:2}}' tag='button'>首页</router-link>

3.    <router-link :to='{name:'myName',params:{c:3,d:4}}' tag='button'>列表1</router-link>

4.    <router-link :to='{path:'/list',params:{c:3,d:4}}' tag='button'>列表2</router-link>

5.    <router-view></router-link>

6.</div>

1.//先引入vue.js & vue-router.js

2.//通过点击按钮获取后台数据，根据后台数据决定去跳转哪个路由

3.const home={

4.    template:'<h2>首页<button @click='fn'>跳转到列表2</button></h2>',

5.    methods:{

6.        //$route：存储的是参数

7.        //$router：存储的是方法

8.        fn(){

9.            //2秒后跳转到列表2

10.            setTimeout(()=>{

11.                //this.$router.go(-x);跳转到之前打开过的页面，跳转X次

12.                //this.$router.back();跳转到前一个打开过的页面

13.                //this.#router.push({path:'/list2',query:{e:5}});从当前页面跳转到列表2，并传参，跳转后再点击浏览器的返回，会回到当前页面

14.                //this.$router.push('/list2');跳转到列表2，不传参时的写法

15.                this.$router.replace({path:'/list2',query:{e:5}});//从当前页面跳转到列表2，并传参，跳转后，再点击浏览器的返回，会直接跳转到当前页面的前一个打开过的页面

16.            },2000)

17.        }

18.    }

19.};

20.const list={

21.    template:'<h2>列表1</h2>',

22.};

23.const list2={

24.    template:'<h2>列表2</h2>',

25.};

26.const routes=[

27.    {

28.        path:'/home',

29.        component:home,

30.    },

31.    {

32.        path:'/list/:c/:d',

33.        name:'myName',

34.        component:list,

35.    },

36.    {

37.        path:'/list2';

38.        component:list2,

39.    }

40.];

41.const router=new VueRouter({

42.    routes,

43.});

44.let vm=new Vue({

45.    el:'#app',

46.    router,

47.})
