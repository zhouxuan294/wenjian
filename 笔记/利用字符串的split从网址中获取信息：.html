利用字符串的split从网址中获取信息：

var str='http://www.zhufengpeixun.com?a=12&b=23';//从中获取’a=12‘和’b=23‘，并保存到对象obj中

function sum(str){

    var obj={};

    var str2=str.split('?')[0];

    var ary=str2.split('&');

    for(var i=0;i<ary.length;i++){

        ary2=ary[i].split('=');

        obj[ary2[0]]=ary2[1];

        }

    return obj;

    }


sum(str);//obj{a:12,b:23}
