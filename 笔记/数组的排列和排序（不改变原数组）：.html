数组的排列和排序（不改变原数组）：

  • toString：不接受参数，返回值是每一项用逗号隔开的字符串。
var ary=[1,2,3,4,5,6];

var  a=ary.toString();

console.log(a);//'1,2,3,4,5,6'

console.log(typeof a);//string

console.log(ary);//[1,2,3,4,5,6]

  • join():接收参数，也可以不给参数；把数组以指定字符连接成字符串。
var ary=[1,2,3,4,5,6];

var a=ary.join();

console.log(a);//'1,2,3,4,5,6'

var b=ary.join(',');

console.log(b);//'1,2,3,4,5,6'

var c=ary.join('+');

console.log(c);//'1+2+3+4+5+6'

var d=ary.join('');

console.log(d);//'123456'

var e=ary.join(' ');

console.log(e);//'1 2 3 4 5 6'


拓展：
1）数组每一项相加求和：

var ary=[1,2,3,4,5,6];

var total=0;

for(var i=0;i<ary.length;i++){

    total+=ary[i];

    }

console.log(total);//21


2）eval():把字符串转换成表达式

var a='1+2+3';

console.log(eval(a));//6

var b='1,2,3';

console.log(eval(b));//3

var c='123';

console.log(eval(c));//123

  • concat()：用于拼接数组，参数可有可无，参数的数据类型不限。
var ary=[1,2,3];

var a=ary.concat();//没有参数时，是负值一份返回给a

console.log(a);//[1,2,3]

console.log(ary);//[1,2,3]

a[0]=9;//返回的新数组a改变，不会影响原数组

console.log(a);//[9,2,3]

console.log(ary);//[1,2,3]

//当参数中有数组时，会先把数组拆开（只拆一层）在与原数组拼接，当参数中有其他数据类型时，是整个与原数组拼接；在原数组后面添加。

var b=ary.concat(12,[8,9],{w:1},function(){console.log(1)});

console.log(b);//[1,2,3,12,8,9,{w:1},f()]

var c=ary.concat(1,[1,[1,2,3],2]);//[1,2,3,1,1,[1,2,3],2]

  • slice(n,m)：从索引n（包含）开始，获取到索引m（不包含）；返回值是获取到的项目组成的新数组。
var ary=[1,2,3,4,5,6];

ary.slice(1,2);//[2]

ary.slice(2,1);//[]

ary.slice();//[1,2,3,4,5,6]

ary.slice(-1,-2);//[]

ary.slice(-2,-1);//[5]

ary.slice(1,1);//[]

ary.slice(0);//[1,2,3,4,5,6]

  • indexOf(xxx,n)：从索引n开始往后查找数组中是否有xxx，若有，返回该项对应的索引值，且不再往后查找，若没有，返回-1；默认n=0。
  • lastIndexOf(xxx,n)：从索引n开始往前查找数组中是否有xxx。，若有，返回该项对应的索引值，且不再往前查找，若没有，返回-1；默认n=0。
var ary=[1,2,3,4,1,6];

ary.indexOf(3);//2

ary.lastIndexOf(3);//2

ary.indexOf(1);//0

ary.lastIndexOf(1);//4

ary.indexOf(1,2);//4

ary.lastIndexOf(1,2);//0

ary.indexOf(8);//-1

ary.lastIndexOf(8);//-1
