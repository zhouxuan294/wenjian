string

  • string（字符串）数据类型，用单引号或者双引号包起来的都是字符串。
  • 字符串的长度length：
var str='www\'qqq';

str.length;//7

  • charAt()：查看对应索引位置的字符。 
var str='abcde';

str.charAt(2);//c

  • charCodeAt()：查看对应索引位置的字符的ASCⅡ值。
var str='abcde';

str.charCodeAt(2);//99

  • toUpperCase()：把小写字母改为大写字母，没有改变原字符串。
var str='abcde';

str.toUpperCase();//'ABCDE'

  • toLowerCase()：把大写字母改为小写字母，没有改变原字符串。
var str='ABCDE';

str.toLowerCase();//abcde

  • indexOf(n,m)：从索引值为m的字符开始往后查找字符n是否在字符串中，若存在，返回n对应的索引值，且不再往后查找，若不存在，返回-1。没有参数m时，默认从索引0开始查找。
  • lastIndexOf(n,m)：从索引值为m的字符开始往前查找字符n是否在字符串中，若存在，返回n对应的索引值，且不再往前查找，若没有，返回-1。
  • split()：把字符串以指定字符拆分成数组。
var str='a,b,c+d,e-f';

str.split(',');//['a','b','c+d','e-f']

str.split('+');//['a,b,c','d,e-f']

str.split('-');//['a,b,c+d,e','f']

  • substr(n,m)：从索引n开始获取m个元素。类似于ary.splice(n,m,x,y,z)。
var str='abcdef';

str.substr(2,1);//'c'

str.substr();//'abcdef'

str.substr(0);//'abcdef'

str.substr(2);//'cdef'

str,substr(-2,1);//'e'

  • substring(n,m)：从索引n开始，获取到索引m（不包含）。当n>m时，两者会自动交换位置。类似于ary.slice(n,m)。 
var str='abcdef';

str.substring(1,3);//'bc'

str.substring(3,1);//'bc'

str.substring();//'abcdef'

str.substring(0);//'abcdef'

  • slice(n,m)：作用同substring。
  • replace('n','m')：把字符串中的n用m代替。
var str='abcdef';

str.replace('c','g');//'abgdef'
