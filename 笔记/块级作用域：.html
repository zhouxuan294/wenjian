块级作用域：

let：

1.for(var i=0;i<10;i++){

2.   console.log(i);

3.}

4.console.log(i);//10

上面这段代码中，因为只想在for循环内部上下文中使用变量 i ，但实际上 i 可以在全局作用域中访问到，污染了整个作用域。而在ES6中，引入了let关键字，提供了除var以外的另一种声明方式。let 关键字可以将变量绑定到所在的作用域中（通常是{}内部），实现块作用域。

1.{let a=1};

2.console,log(a);//ReferenceError: a is not defined

块作用域实际上可以替代立即执行匿名函数（IIFE）。

1.(function(){

2.   var a=2;

3.})();

4.console.log(a);//ReferenceError: a is not defined

for循环中的变量用let声明，将会避免作用域污染问题：

1.for(let a=0;a<10;a++){

2.   console.log(a);

3.}

4.console.log(a);//ReferenceError: a is not defined

5.//--上下两段代码等价--

6.{

7.   let j;

8.   for(j=0;j<10;j++){

9.      let i=j;

10.      console.log(i);

11.   }

12.}

下面代码中，由于闭包只能取得包含函数中的任何变量的最后一个值，所以控制台输出5而不是0（也可以理解为：除了自执行函数，其他函数都需要调用才会执行，所以下面的代码中，在for循环执行过程中，直到结束，其中的函数也不会执行，因此函数执行时，返回的只有5）。

1.var a=[];

2.for(var i=0;i<5;i++){

3.   a[i]=function(){

4.      return i;

5.   }

6.}

7.console.log(a[0]());//5

可以通过函数传参来确保每次循环的值：

1.var a=[];

2.for(var i=0;i<5;i++){

3.   a[i]=(function(n){

4.      return function(){

5.         return n;

6.      };

7.   })(i);

8.};

9.console.log(a[0]());//0

使用let更方便，由于let循环有一个重新赋值的过程，相当于保存了每一次循环时的值。用var定义变量的时候，在整个循环中是全局的，每一次循环实际上是为i赋值，循环一次就赋值一次，5次循环完，i最后的结果赋值就是5；let的作用域是块级作用域，也就是每次循环的i的作用域就是本次循环，下次循环重新定义变量i：

1.var a=[];

2.for(let i=0;i<5;i++){

3.   a[i]=function(){

4.      return i;

5.   }

6.}

7.console.log(a[0]());//0

用let声明的变量，不能重复声明，也不会在块作用域中进行提升

const：

出了let以外，ES6还引入了const，同样可以用来创建块作用域变量，但其值是固定的（常量），之后任何试图修改值的操作都会引起错误。

1.if(true){

2.   var a=1;

3.   const b=2;

4.   a=3;

5.   b=4;//Assignment to constant variable

6.}

const声明的常量，与let一样不可重复声明。

try：

try-catch语句的一个常见用途就是创建块级作用域，其中声明的变量仅仅在catch内部有效

1.try{

2.   throw 2;

3.}catch(a){

4.   console.log(a);//2

5.}

6.console.log(a);//ReferenceError: a is not defined

7.//--上下两段代码等价--

8.try{

9.   throw undefined;

10.}catch(a){

11.   a=2;

12.   console.log(a);//2

13.}

14.console.log(a);//ReferenceError: a is not defined
