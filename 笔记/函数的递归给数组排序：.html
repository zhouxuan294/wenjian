函数的递归给数组排序：

function sum(ary){

    if(ary.length<=1){

        return ary;

        }

    var n=parseInt(ary.length/2),//数组中间项的索引值

        temp=ary.splice(n,1)[0],

        left=[],

        right=[];

    for(var i=0;i<ary.length;i++){

        if(temp>=ary[i]){

            left.push(ary[i]);

            }else{

            right.push(ary[i]);

            }

        }

    return sum(left).concat(temp,sum(right));

    }


上述代码为从小到大排序，若要从大到小排序，将if(temp>=ary[i])改为if(temp<=ary[i])即可。
