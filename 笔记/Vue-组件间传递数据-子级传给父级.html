Vue-组件间传递数据-子级传给父级

1.<div id='app'>

2.    <demo_2></demo_2>

3.</div>

1.<!--子级组件-->

2.<template id='test_1'>

3.    </div>

4.        <button @click='fn_1'>子级组件的按钮</button>

5.        {{n}}

6.        {{a}}

7.        {{o}}

8.        {{s}}

9.    </div>

10.</template>

11.<!--父级组件-->

12.<template id='test_2'>

13.    <div>

14.        <button>父级组件的按钮</button>

15.        {{N}}

16.        {{A}}

17.        {{O}}

18.        {{S}}

19.        <demo_1 @fn_1-1='fn_2'></demo_1><!--" @fn_1-1 " 是自定义函数-->

20.    </div>

21.</template>

1.//———子级组件———

2.let demo_1={

3.    template:'#test_1',

4.    data(){

5.        return {

6.            n:1,

7.            a:[1,2,3],

8.            o:{name:'DEMO'},

9.            s:'string'

10.        }

11.    },

12.    methods:{

13.        fn_1(){

14.            this.$emit('fn_1-1',this.n,this.a,this.o,this.s);

15.        }

16.    }

17.};

18.//———父级组件———

19.let demo_2={

20.    template:'#test_2',

21.    data(){

22.        return {

23.            N:0,

24.            A:[],

25.            O:{},

26.            S:'_'

27.        }

28.    },

29.    methods:{

30.        fn_2(){

31.            console.log(arguments);//是子级组件通过自定义函数传递过来的数据，详细见下面的截图

32.            this.N=arguments[0];

33.            this.A=arguments[1];

34.            this.O=arguments[2];

35.            this.S=arguments[3];

36.        }

37.    },

38.    components:{

39.        demo_1,

40.    }

41.};

42.let vm=new Vue({

43.    el:'app',

44.    components:{

45.        demo_2,

46.    }

47.})

注意：如果子级组件传给父级组件的是引用数据类型的时候，若父级组件修改了该数据那么子级组件的元数据也会被修改
